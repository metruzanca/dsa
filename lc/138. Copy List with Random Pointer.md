```js
function copyRandomList(head) {
    const map = new Map()
    let current = head
    while (current !== null) {
        map.set(current, {val: current.val, next: null, random: null})
        current = current.next
    }

    current = head
    while (current !== null) {
        const node = map.get(current)
        node.next = map.get(current.next)
        node.random = map.get(current.random)
        current = current.next
    }
    return map.get(head)
};
```
